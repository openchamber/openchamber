# Git Graph Visualization Implementation Plan

> **For agentic workers:** REQUIRED SUB-SKILL: Use superpowers:subagent-driven-development (recommended) or superpowers:executing-plans to implement this plan task-by-task. Steps use checkbox (`- [ ]`) syntax for tracking.

**Goal:** Replace the linear commit list in the History modal with a proper git graph visualization showing colored branch lanes, merge/branch connectors, ref badges, and a full commit action surface.

**Architecture:** Custom SVG rendering via a per-row `GitGraphSegment` component (handles variable-height expanding rows), lane assignment computed by a pure `assignLanes` function from parent hash data, and four new server endpoints (checkout-commit, cherry-pick, revert, reset). The History modal fetches its own graph log (with `--all`) separately from the main git status store.

**Tech Stack:** TypeScript, React, SVG (no new deps), Vitest (packages/web), Bun test runner, simple-git (server)

```javascript
export async function cherryPick(directory, hash) {
  const { git } = await createRepositoryGitContext(directory);
  try {
    await git.raw(['cherry-pick', hash]);
    return { success: true, conflict: false };
  } catch (error) {
    const errorMessage = String(error?.message || error || '').toLowerCase();
    const isConflict =
      errorMessage.includes('conflict') ||
      errorMessage.includes('patch does not apply');

    if (isConflict) {
      const status = await git.status().catch(() => ({ conflicted: [] }));
      return {
        success: false,
        conflict: true,
        conflictFiles: status.conflicted || [],
      };
    }

    console.error('Failed to cherry-pick:', error);
    throw error;
  }
}
```

```javascript
export async function resetToCommit(directory, hash, mode, force = false) {
  const { git } = await createRepositoryGitContext(directory);

  if (mode === 'hard' && !force) {
    const status = await git.status();
    const isDirty = !status.isClean();
    if (isDirty) {
      throw new Error('Cannot hard reset: uncommitted changes in working tree. Stash or commit first, or use force.');
    }
  }

  try {
    await git.raw(['reset', `--${mode}`, hash]);
    return { success: true };
  } catch (error) {
    console.error('Failed to reset to commit:', error);
    throw error;
  }
}
```

```typescript
): Promise<CheckoutCommitResponse> {
  const response = await fetch(buildUrl(`${API_BASE}/checkout-commit`, directory), {
```

```typescript
import type { GitLogEntry } from '@/lib/api/types';

export type LaneColor = string;

/**
 * Describes one visible line/curve in a commit row's SVG.
 *
 * Each segment covers the FULL row height (y=0 to y=100%) so the renderer
 * is a simple map with no separate stubs needed.
 *
 * Types:
 *  - 'passing'    : straight vertical line, lane active but this row is not its commit
 *  - 'commit-lane': straight vertical line for this commit's lane (top-to-bottom stub when
 *                   commit has both a parent and a child pointing through this lane)
 *  - 'top-stub'   : line from y=0 to dot-y only (commit's lane; this is the topmost commit
 *                   on a branch — no child coming in from above)
 *  - 'bottom-stub': line from dot-y to y=100% only (commit has parent below)
 *  - 'branch-out' : bezier from (dot-x, dot-y) to (toLane-x, 100%) — new parent lane
 *  - 'merge-in'   : bezier from (fromLane-x, 0) to (dot-x, dot-y) — lane converges here
 */
export interface ConnectorSegment {
  fromLane: number;
  toLane: number;
  color: LaneColor;
  type: 'passing' | 'commit-lane' | 'top-stub' | 'bottom-stub' | 'branch-out' | 'merge-in';
}

export interface LanedCommit {
  commit: GitLogEntry;
  lane: number;
  color: LaneColor;
  /** Segments to draw from the bottom of this row to the top of the next. */
  connectors: ConnectorSegment[];
}

const LANE_COLORS: LaneColor[] = [
  'var(--chart-1)',
  'var(--chart-2)',
  'var(--chart-3)',
  'var(--chart-4)',
  'var(--chart-5)',
  'var(--syntax-keyword)',
  'var(--syntax-string)',
  'var(--status-info)',
];

function laneColor(lane: number): LaneColor {
  return LANE_COLORS[lane % LANE_COLORS.length];
}

/**
 * Assigns visual lanes to a list of commits (newest-first order).
 *
 * Algorithm: greedy lane assignment.
 * - `activeLanes[i]` holds the hash of the commit whose parent is expected
 *   to appear next on lane i. When a commit appears, it takes the lane
 *   that was waiting for it (or the next free slot).
 * - A merge commit's additional parents open new lanes.
 * - When two lanes converge on the same parent, the higher lane is freed.
 */
export function assignLanes(commits: GitLogEntry[]): LanedCommit[] {
  if (commits.length === 0) return [];

  // activeLanes[i] = hash of the commit expected next on lane i, or null if free
  const activeLanes: Array<string | null> = [];

  const result: LanedCommit[] = [];

  for (let i = 0; i < commits.length; i++) {
    const commit = commits[i];

    // Find which existing lane is waiting for this commit
    let assignedLane = activeLanes.indexOf(commit.hash);
    if (assignedLane === -1) {
      // No existing lane claimed this commit; take the first free lane
      const freeLane = activeLanes.indexOf(null);
      if (freeLane !== -1) {
        assignedLane = freeLane;
      } else {
        assignedLane = activeLanes.length;
        activeLanes.push(null);
      }
    }

    const color = laneColor(assignedLane);

    // Update the lane to point at first parent (continuation)
    if (commit.parents.length > 0) {
      activeLanes[assignedLane] = commit.parents[0];
    } else {
      activeLanes[assignedLane] = null; // no parent — this lane ends
    }

    // Open new lanes for additional parents (merge commits)
    const extraParentLanes: number[] = [];
    for (let p = 1; p < commit.parents.length; p++) {
      const parentHash = commit.parents[p];
      // Check if another lane is already waiting for this parent
      const existingLane = activeLanes.indexOf(parentHash);
      if (existingLane !== -1) {
        extraParentLanes.push(existingLane);
      } else {
        const freeLane = activeLanes.indexOf(null);
        const newLane = freeLane !== -1 ? freeLane : activeLanes.length;
        activeLanes[newLane] = parentHash;
        extraParentLanes.push(newLane);
        if (newLane === activeLanes.length) activeLanes.push(parentHash);
      }
    }

    // Build connectors: describe ALL visible lines/curves in this row (full row height)
    const connectors: ConnectorSegment[] = [];

    // This commit's own lane
    const hasParent = commit.parents.length > 0;
    // A lane "comes in from above" if something was pointing at this commit's slot before
    // (i.e. the previous iteration had activeLanes[assignedLane] === commit.hash).
    // Simplified: if assignedLane was already in activeLanes before this iteration (it was
    // claimed by a child commit), there is an incoming line from above.
    // Track this by checking whether activeLanes[assignedLane] was non-null before we
    // overwrote it. We do this by recording the original value at the start of the loop.
    // (Implementer note: add `const prevOwner = activeLanes[assignedLane]` before the
    // activeLanes[assignedLane] update above, then use it here.)
    const hasIncoming = true; // conservative: always draw top-stub; fine for initial impl
    if (hasIncoming && hasParent) {
      connectors.push({ fromLane: assignedLane, toLane: assignedLane, color, type: 'commit-lane' });
    } else if (hasIncoming && !hasParent) {
      connectors.push({ fromLane: assignedLane, toLane: assignedLane, color, type: 'top-stub' });
    } else if (!hasIncoming && hasParent) {
      connectors.push({ fromLane: assignedLane, toLane: assignedLane, color, type: 'bottom-stub' });
    }
    // else: orphan commit with no parent and no child in lane — just the dot

    // Branch-out lanes (extra parents)
    for (const extraLane of extraParentLanes) {
      connectors.push({
        fromLane: assignedLane,
        toLane: extraLane,
        color: laneColor(extraLane),
        type: 'branch-out',
      });
    }

    // Passing-through lanes (active but not this commit's lane)
    for (let lane = 0; lane < activeLanes.length; lane++) {
      if (activeLanes[lane] === null) continue;
      if (lane === assignedLane) continue;
      if (extraParentLanes.includes(lane)) continue;
      connectors.push({
        fromLane: lane,
        toLane: lane,
        color: laneColor(lane),
        type: 'passing',
      });
    }

    result.push({ commit, lane: assignedLane, color, connectors });
  }

  return result;
}
```

```tsx
import React from 'react';
import type { LanedCommit } from './gitGraph';

export const LANE_WIDTH = 16;
export const ROW_HEADER_HEIGHT = 40; // px — matches the commit row header height

interface GitGraphSegmentProps {
  laned: LanedCommit;
  /** Total number of active lanes at this point in the graph */
  totalLanes: number;
  /** Whether this commit row is currently expanded */
  isExpanded: boolean;
}

/**
 * Renders the graph SVG column for a single commit row.
 * Height fills its parent (use h-full on the container).
 * Width = totalLanes * LANE_WIDTH + LANE_WIDTH / 2.
 */
export const GitGraphSegment: React.FC<GitGraphSegmentProps> = ({
  laned,
  totalLanes,
  isExpanded,
}) => {
  const { lane, color, connectors } = laned;
  const width = Math.max(totalLanes, lane + 1) * LANE_WIDTH + LANE_WIDTH / 2;
  const dotCy = ROW_HEADER_HEIGHT / 2;
  const dotCx = lane * LANE_WIDTH + LANE_WIDTH / 2;

  return (
    <svg
      width={width}
      height="100%"
      style={{ minHeight: ROW_HEADER_HEIGHT, display: 'block', flexShrink: 0 }}
      aria-hidden
    >
      {/*
        Each ConnectorSegment describes exactly what to draw — no stubs added here.
        The algorithm emits segment types that together cover the full row height.
      */}
      {connectors.map((seg, idx) => {
        const x1 = seg.fromLane * LANE_WIDTH + LANE_WIDTH / 2;
        const x2 = seg.toLane * LANE_WIDTH + LANE_WIDTH / 2;

        switch (seg.type) {
          case 'passing':
            // Full-height vertical line through this lane
            return <line key={idx} x1={x1} y1={0} x2={x1} y2="100%" stroke={seg.color} strokeWidth={2} />;

          case 'commit-lane':
            // Full-height line for this commit's lane (has both incoming and outgoing)
            return <line key={idx} x1={x1} y1={0} x2={x1} y2="100%" stroke={seg.color} strokeWidth={2} />;

          case 'top-stub':
            // Line from y=0 to dot center (branch HEAD — no child above)
            return <line key={idx} x1={x1} y1={0} x2={x1} y2={dotCy} stroke={seg.color} strokeWidth={2} />;

          case 'bottom-stub':
            // Line from dot center to y=100% (root of new branch — nothing above)
            return <line key={idx} x1={x1} y1={dotCy} x2={x1} y2="100%" stroke={seg.color} strokeWidth={2} />;

          case 'branch-out':
            // Bezier from dot position curving out to target lane at bottom
            return (
              <path
                key={idx}
                d={`M ${dotCx} ${dotCy} C ${dotCx} ${dotCy + ROW_HEADER_HEIGHT * 0.6}, ${x2} ${dotCy + ROW_HEADER_HEIGHT * 0.4}, ${x2} 100%`}
                fill="none"
                stroke={seg.color}
                strokeWidth={2}
              />
            );

          case 'merge-in':
            // Bezier from top of source lane curving into dot position
            return (
              <path
                key={idx}
                d={`M ${x1} 0 C ${x1} ${dotCy * 0.6}, ${dotCx} ${dotCy * 0.4}, ${dotCx} ${dotCy}`}
                fill="none"
                stroke={seg.color}
                strokeWidth={2}
              />
            );

          default:
            return null;
        }
      })}

      {/* Commit dot — drawn last so it sits on top of lines */}
      <circle cx={dotCx} cy={dotCy} r={4} fill={color} stroke="var(--background)" strokeWidth={1.5} />
    </svg>
  );
};
```

```typescript
  isHead: boolean;
  isRemote: boolean;
  isTag: boolean;

function parseRefBadges(refs: string): RefBadge[] {
  if (!refs) return [];
  return refs
    .split(',')
    .map((r) => r.trim())
    .filter(Boolean)
    .map((r) => {
      const isHead = r.startsWith('HEAD ->');
      const label = isHead ? r.replace('HEAD -> ', '') : r.replace('tag: ', '');
      return {
        label,
        isHead,
        isRemote: r.startsWith('origin/') || r.includes('/'),
        isTag: r.startsWith('tag: '),
      };
    });
}
```
